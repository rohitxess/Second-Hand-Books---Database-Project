create table customer 
(
	login_ID varchar(20), 
	name varchar(50) not null ,
	email_ID varchar(50) not null unique,
	age integer,
	address varchar(100) not null,
	stateno integer not null,
	pincode integer not null,
	phone_number char(12) not null unique,
	

		constraint customerPK primary key (login_ID),								-- Primary Key

		constraint customer_login_ID check (login_ID like 'clog%'),

		constraint customer_age check ( (age>=8) and (age<122) ),						-- Longest living persons age is 122	

		constraint customer_email_ID check ( email_ID like ('%@%.%') ),						-- To check valid email

		constraint customer_stateno check ( (stateno >= 1 ) and (stateno <=29 ) ),				-- To check the number of states in India

		constraint customer_phone_number check ( phone_number like ( '9%' ) or phone_number like ( '8%' ) or phone_number like ('7%' ) or phone_number like ('____-_______' ) )			-- To check whether the phone number is starting from the right digit
);



create table shop
(
	shop_ID varchar(20),
	address varchar(100) not null,
	stateno integer not null,
	pincode integer not null,
	phone_number char(12) not null unique,  
	shipping text,

		constraint shopPK primary key (shop_ID),							-- Primary Key 

		constraint shop_shipping check ( shipping in ( 'yes' , 'no' ) ),				-- Constraint to check shipping availabilty 
	
		constraint store_emp_store_ID check (shop_ID like 'shpid%'),				 	-- constraint for shop id

		constraint shop_stateno check ( (stateno>=1) and (stateno<=29 ) ),				-- Constraint to check the number of states

		constraint shop_phone_number check ( phone_number like ( '9%' ) or phone_number like ( '8%' ) or phone_number like ('7%' ) or phone_number like ('____-_______' ) )			-- To check whether the phone number is starting from the right digit
);

create table book 
(

	book_isbn char(20) not null,			-- Primary Key, Foreign Key 
	book_name text not null,	
	category varchar(20),
	bestselleryear integer,				-- default entry for bestselleryear will be NULL no need to set it to null
	edition varchar(20) default 'Not Available' ,
	discount text,

		constraint bookPK primary key (book_isbn),

		constraint book_category check(		
		      		category in ( 'Fiction',	
					       'Poetry',
						'Non Fictional',
						'Mystery',
						'Comic',
						'Fairy Tale',
						'Thriller',
						'Fantasy',
						'Adventure',
						'Crime' , 'history', 'Geography', 'Mathematics', 'Biology', 'Science', 'Computer Science')),										-- Categories of books available in shops 

	
		constraint book_bestselleryear check ( (bestselleryear>=1900) and (bestselleryear<=2019)),				-- Constraint to check the best seller year  /* bestselleryear between 1990 and 2019 */

		constraint book_edition check ( edition in (2,3,4) or ( ( edition = 'Oldest' ) or ( edition = 'Latest') ) ),		-- Constraint to check the edition of the book

		constraint book_discount check ( discount in ( 'yes' , 'no') ) 							-- Constraint for checking the avilablity of discount
);


create table bookshop
(
	shop_ID varchar(20),
	book_isbn char(20),
	
		
		constraint bookshopPK primary key (book_isbn,shop_ID),

		constraint bookshopFK foreign key (book_isbn) references book,

		constraint bookshopFK1 foreign key (shop_ID) references shop on delete cascade

);



create table store_emp
(

	emp_ID varchar(20), 												-- primary key on delete restrict
	empname varchar(50) not null ,
	designation varchar(20) not null,
	address varchar(100) not null,												
	phone_number char(12) unique,
	shop_ID varchar(20),

		constraint store_empPK primary key (emp_ID),

		constraint store_empFK1 foreign key(shop_ID) references shop on delete cascade,

		constraint store_emp_emp_ID check (emp_ID like 'empid%'), 				 		-- constraint for employee id

		constraint store_emp_designation check( 
			designation in ( 'Owner',											-- Owner of the Shop
					 'Cashier',											-- Employee which handles the cash counter
					 'Partner',											-- Partner with Owner
					 'Manager',											-- Employees designated for cleaning
				         'Front-BD',     										-- Employees dealing with coustomers
					 'Back-BD')),											-- Employees handling books in the warehouse 
					  

		constraint customer_phone_number check ( phone_number like ( '9%' ) or phone_number like ( '8%' ) or phone_number like ('7%' ) or phone_number like ('____-_______' ) )			-- To check whether the phone number is starting from the right digit
);

-- View for Customer clog1 who purchased some books

CREATE VIEW customerpurchase AS
    SELECT name as Customername, email_ID as CustomerEmail , stateno as CustomerState,
           order_ID as CustomerOrder, total_price as CustomerPrice
    FROM customer natural join productorder
    WHERE login_ID='clog1';


-- View for books having more than 1 author

CREATE VIEW manybookauthor AS
    select book_isbn as BookNumber, count(*) as NumberOfAuthors
	from author natural join bookauthor group by book_isbn having
	 count(*)>1 order by count(*);
